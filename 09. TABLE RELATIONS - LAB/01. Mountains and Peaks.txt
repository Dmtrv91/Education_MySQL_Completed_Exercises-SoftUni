USE camp;

CREATE TABLE mountains(
id INT(11) AUTO_INCREMENT PRIMARY KEY NOT NULL,
name VARCHAR(50) NOT NULL
);

CREATE TABLE peaks(
id INT(11) AUTO_INCREMENT PRIMARY KEY NOT NULL,
name VARCHAR(50) NOT NULL,
mountain_id INT(11) NOT NULL,
CONSTRAINT fk_peaks_mountains 
FOREIGN KEY(mountain_id)
REFERENCES mountains(id)
);
